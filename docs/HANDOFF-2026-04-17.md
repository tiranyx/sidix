### Error yang Sudah Diperbaiki:
| Error | Root Cause | Fix |
|-------|-----------|-----|
| `SFTConfig ImportError` | trl 0.8.6 tidak ada SFTConfig | `from transformers import TrainingArguments as SFTConfig` |
| `dataset_text_field TypeError` | newer trl auto-detect | Hapus param `dataset_text_field` |
| `tokenizer kwarg TypeError` | renamed di trl baru | Ganti `tokenizer=` → `processing_class=` |
| `numpy binary incompatibility` | pip downgrade numpy | Jangan install numpy/pandas, hanya install 4 package di atas |
| `OutOfMemoryError CUDA` | model lama masih di VRAM | Restart kernel dulu, lalu `low_cpu_mem_usage=True` |
| `NotImplementedError BFloat16` | `paged_adamw_32bit` + fp16 conflict | Ganti ke `adamw_torch`, `fp16=False` |

```python
# Di agent_serve.py, ganti _llm_generate():
from transformers import AutoModelForCausalLM, AutoTokenizer
from peft import PeftModel
import torch

# Load saat startup (tambahkan sebelum create_app())
_model = None
_tokenizer = None

def _load_model():
    global _model, _tokenizer
    adapter_path = Path("models/sidix-lora-adapter")
    base_model_name = "Qwen/Qwen2.5-7B-Instruct"
    _tokenizer = AutoTokenizer.from_pretrained(str(adapter_path))
    base = AutoModelForCausalLM.from_pretrained(
        base_model_name,
        torch_dtype=torch.float16,
        device_map="auto",
        low_cpu_mem_usage=True,
    )
    _model = PeftModel.from_pretrained(base, str(adapter_path))
    _model.eval()

def _llm_generate(prompt, system, max_tokens=256, temperature=0.7):
    if _model is None:
        return "Model belum di-load.", "mock"
    messages = [{"role": "system", "content": system}, {"role": "user", "content": prompt}]
    text = _tokenizer.apply_chat_template(messages, tokenize=False, add_generation_prompt=True)
    inputs = _tokenizer([text], return_tensors="pt").to(_model.device)
    with torch.no_grad():
        out = _model.generate(**inputs, max_new_tokens=max_tokens, temperature=temperature, do_sample=True)
    generated = out[0][inputs["input_ids"].shape[-1]:]
    return _tokenizer.decode(generated, skip_special_tokens=True), "local_lora"
```
